1. Program to create a class named Validator and define a static method that checks whether a password is strong

Ans:
class Validator:
    @staticmethod
    def is_strong_password(password):
        if len(password) < 8:
            return False
        if not any(char.isdigit() for char in password):
            return False
        if not any(char.isupper() for char in password):
            return False
        if not any(char.islower() for char in password):
            return False
        return True

# Example usage
pwd = "Strong123"
print("Strong Password:", Validator.is_strong_password(pwd))

2. Program where a static method converts temperature from Celsius to Fahrenheit and Fahrenheit to Celsius

Ans:
class TemperatureConverter:
    @staticmethod
    def celsius_to_fahrenheit(c):
        return (c * 9 / 5) + 32

    @staticmethod
    def fahrenheit_to_celsius(f):
        return (f - 32) * 5 / 9

# Example usage
print("30°C in Fahrenheit:", TemperatureConverter.celsius_to_fahrenheit(30))
print("86°F in Celsius:", TemperatureConverter.fahrenheit_to_celsius(86))

Ans:
class Utility:
    @staticmethod
    def is_leap_year(year):
        if (year % 400 == 0) or (year % 4 == 0 and year % 100 != 0):
            return True
        return False
# Example usage

4. Program to validate a product code using a static method with a predefined format

Ans:
Format: PROD-XXXX (PROD followed by 4 digits)
class ProductValidator:
    @staticmethod
    def validate_code(code):
        if len(code) != 9:
            return False
        if not code.startswith("PROD-"):
            return False
        if not code[5:].isdigit():
            return False
        return True

# Example usage
code = "PROD-1234"
print("Valid Product Code:", ProductValidator.validate_code(code))

5. Program that demonstrates reuse of a static method across two different classes

Ans:
class MathUtils:
    @staticmethod
    def square(num):
        return num * num

class AreaCalculator:
    def square_area(self, side):
        return MathUtils.square(side)

class VolumeCalculator:
    def cube_volume(self, side):
        return MathUtils.square(side) * side

# Example usage
area = AreaCalculator()
volume = VolumeCalculator()
print("Area of square:", area.square_area(5))
print("Volume of cube:", volume.cube_volume(5))

7. Program where a class method works as a factory method to create objects from dictionary data

Ans:
class Student:
    def __init__(self, name, age, marks):
        self.name = name
        self.age = age
        self.marks = marks

    @classmethod
    def from_dict(cls, data):
        return cls(data["name"], data["age"], data["marks"])

# Example usage
data = {"name": "Alok", "age": 21, "marks": 88}
student = Student.from_dict(data)
print(student.name, student.age, student.marks)

8. Program to create a Bank class where the interest rate is shared and updated using a class method

Ans:
class Bank:
    interest_rate = 5.0

    def __init__(self, balance):
        self.balance = balance

    @classmethod
    def update_interest_rate(cls, new_rate):
        cls.interest_rate = new_rate

    def calculate_interest(self):
        return self.balance * Bank.interest_rate / 100

# Example usage
acc1 = Bank(10000)
acc2 = Bank(20000)
Bank.update_interest_rate(6.5)
print("Interest 1:", acc1.calculate_interest())
print("Interest 2:", acc2.calculate_interest())

14. Program to demonstrate name mangling behavior of private methods

Ans:
class Demo:
    def __show(self):
        print("Private method called")

    def access(self):
        self.__show()

# Example usage
d = Demo()
d.access()

# Accessing private method using name mangling
d._Demo__show()
Explanation:
Python internally renames __show to _Demo__show.
This is called name mangling.

15. Program where changes in a private method do not affect external code

Ans:
class Calculator:
    def add(self, a, b):
        return self.__calculate(a, b)

    def __calculate(self, a, b):
        return a + b

# Example usage
calc = Calculator()
print(calc.add(5, 3))
Explanation:
Even if the internal logic of __calculate() is changed later,
external code calling add() will not be affected, ensuring data hiding and security.
